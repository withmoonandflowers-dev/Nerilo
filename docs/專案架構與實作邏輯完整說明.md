```typescript
// 用戶操作：輸入訊息並點擊「傳送」
handleSend() {
  // 步驟 1: GossipMessageHandler.sendMessage()
  // 1.1 Rate limiting 檢查
  if (!checkSendRate(userId)) {
    throw new Error('Rate limit exceeded');
  }
  
  // 1.2 序列號遞增
  seq++;
  
  // 1.3 建立 GossipMessage（不含簽名）
  const message: Omit<GossipMessage, 'signature'> = {
    roomId,
    senderId: userId,
    pubKey: await exportPublicKey(),
    seq,
    timestamp: Date.now(),
    content: inputValue,
    ttl: 8,
  };
  
  // 1.4 簽名訊息
  const signature = await securityManager.signMessage(
    message,
    identityManager.getPrivateKey()
  );
  
  const signedMessage: GossipMessage = { ...message, signature };
  
  // 1.5 隨機選 2 個鄰居發送
  const neighbors = topologyManager.getNeighbors();
  const selected = selectRandomNeighbors(neighbors, 2);
  for (const neighbor of selected) {
    await neighbor.send(signedMessage);
  }
  
  // 1.6 記錄已發送（去重）
  const messageId = await getMessageId(signedMessage);
  seenMessageIds.add(messageId);
  
  // 步驟 2: 鄰居接收訊息
  // 2.1 檢查是否已見過（去重）
  if (seenMessageIds.has(messageId)) return;
  
  // 2.2 驗證簽名
  const publicKey = await importPublicKey(message.pubKey);
  const isValid = await verifyMessage(message, publicKey);
  if (!isValid) return;
  
  // 2.3 檢查序列號（防止重放）
  if (!checkSequence(message.senderId, message.seq)) return;
  
  // 2.4 檢查 TTL
  if (message.ttl <= 0) return;
  
  // 2.5 顯示訊息
  notifyMessageListeners(message);
  
  // 2.6 轉發（如果 TTL > 0）
  if (message.ttl > 0) {
    message.ttl -= 1;
    await forwardMessage(message, fromNeighbor);
  }
}
```

```typescript
class MessageBatcher {
  private batch: ChatMessage[] = [];
  private batchSize = 10;
  private flushTimeout = 1000;
  
  add(message: ChatMessage) {
    this.batch.push(message);
    if (this.batch.length >= this.batchSize) {
      this.flush();
    } else {
      this.scheduleFlush();
    }
  }
  
  private scheduleFlush() {
    setTimeout(() => this.flush(), this.flushTimeout);
  }
  
  private async flush() {
    // 批次儲存到 IndexedDB
    await indexedDBService.saveChatMessages(this.batch, roomId);
    this.batch = [];
  }
}
```

```typescript
class PublicKeyCache {
  private cache = new Map<string, { key: CryptoKey; timestamp: number }>();
  private readonly TTL = 60 * 60 * 1000; // 1 小時
  
  async get(userId: string): Promise<CryptoKey> {
    const cached = this.cache.get(userId);
    if (cached && Date.now() - cached.timestamp < this.TTL) {
      return cached.key;
    }
    
    // 從 Firestore 載入
    const room = await RoomService.getRoom(roomId);
    const identity = room?.meshIdentities?.[firebaseUid];
    if (!identity) {
      throw new Error('Identity not found');
    }
    
    const key = await securityManager.importPublicKey(identity.pubKey);
    this.cache.set(userId, { key, timestamp: Date.now() });
    return key;
  }
}
```

```typescript
  private static getUserFriendlyMessage(error: Error): string {
    if (error.message.includes('network')) {
      return '網路連線問題，請檢查您的網路';
    }
    if (error.message.includes('permission')) {
      return '權限不足，請確認您已登入';
    }
    return '發生未知錯誤，請稍後再試';
  }
}
```

### 2. 重試機制

```typescript
async function withRetry<T>(
  fn: () => Promise<T>,
  maxRetries: number = 3,
  delay: number = 1000
): Promise<T> {
  for (let i = 0; i < maxRetries; i++) {
    try {
      return await fn();
    } catch (error) {
      if (i === maxRetries - 1) throw error;
      await new Promise(resolve => setTimeout(resolve, delay * (i + 1)));
    }
  }
  throw new Error('Max retries exceeded');
}

// 使用範例
const room = await withRetry(
  () => RoomService.getRoom(roomId, true),
  3,
  1000
);
```

```typescript
      },

// 效能監控
class PerformanceMonitor {
  private static metrics: Map<string, number[]> = new Map();
  
  static measure(name: string, fn: () => Promise<void>) {
    const start = performance.now();
    return fn().then(() => {
      const duration = performance.now() - start;
      const metrics = this.metrics.get(name) || [];
      metrics.push(duration);
      this.metrics.set(name, metrics);
      
      Logger.info('Performance', {
        name,
        duration: `${duration.toFixed(2)}ms`,
        average: `${(metrics.reduce((a, b) => a + b, 0) / metrics.length).toFixed(2)}ms`,
      });
    });
  }
}
```
